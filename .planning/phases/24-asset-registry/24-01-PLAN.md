#define NT_PACK_MAGIC 0x4B41504E
#define NT_PACK_VERSION 1
#define NT_PACK_VERSION_MAX 1
#define NT_PACK_ASSET_ALIGN 4
#define NT_PACK_DATA_ALIGN 8
#define NT_PACK_ALIGN_UP(size, align) (((size) + ((align) - 1)) & ~((align) - 1))

typedef enum {
    NT_ASSET_MESH = 1,
    NT_ASSET_TEXTURE = 2,
    NT_ASSET_SHADER_CODE = 3,
} nt_asset_type_t;

#pragma pack(push, 1)
typedef struct {
    uint32_t magic;       /* NT_PACK_MAGIC ("NPAK") */
    uint32_t pack_id;     /* unique pack identifier */
    uint16_t version;     /* NT_PACK_VERSION */
    uint16_t asset_count; /* number of assets */
    uint32_t header_size; /* offset where data region starts */
    uint32_t total_size;  /* total file size in bytes */
    uint32_t checksum;    /* CRC32 of data after header */
} NtPackHeader;  // 24 bytes
#pragma pack(pop)

#pragma pack(push, 1)
typedef struct {
    uint32_t resource_id;    /* hash of asset path */

    uint8_t asset_type;      /* nt_asset_type_t */
    uint8_t _pad;
    uint32_t offset;         /* byte offset from file start */
    uint32_t size;           /* asset data size in bytes */
} NtAssetEntry;  // 16 bytes
#pragma pack(pop)

```c
#define FNV1A_OFFSET_BASIS 0x811C9DC5U
#define FNV1A_PRIME 0x01000193U
uint32_t nt_builder_fnv1a(const char *str) {
    if (!str) return FNV1A_OFFSET_BASIS;
    uint32_t hash = FNV1A_OFFSET_BASIS;
    for (const char *p = str; *p != '\0'; p++) {
        hash ^= (uint8_t)*p;
        hash *= FNV1A_PRIME;
    }
    return hash;
}
```

    **engine/resource/nt_resource.c:** Implementation with:
    - Static module state struct `s_resource` containing:
      - `NtPackMeta packs[NT_RESOURCE_MAX_PACKS]`
      - `NtAssetMeta assets[NT_RESOURCE_MAX_ASSETS]`
      - `NtResourceSlot slots[NT_RESOURCE_MAX_SLOTS + 1]` (index 0 reserved)
      - `uint16_t generations[NT_RESOURCE_MAX_SLOTS + 1]`
      - `uint16_t free_queue[NT_RESOURCE_MAX_SLOTS]`
      - `uint16_t queue_top`
      - `uint32_t asset_count` (next free index in assets[])
      - `uint32_t placeholder_handles[4]` (indexed by asset_type, 0=unused)
      - `bool needs_resolve`
      - `bool initialized`
    - `nt_resource_init()`: memset to zero, fill free_queue with indices NT_RESOURCE_MAX_SLOTS down to 1, set queue_top = NT_RESOURCE_MAX_SLOTS, initialized = true
    - `nt_resource_shutdown()`: memset to zero (static arrays, no free needed)
    - `nt_resource_hash()`: FNV-1a, identical algorithm to `nt_builder_fnv1a()` -- constants `NT_FNV1A_OFFSET_BASIS 0x811C9DC5U`, `NT_FNV1A_PRIME 0x01000193U`, NULL returns offset basis, iterate bytes XOR then multiply
    - `nt_resource_parse_pack()`:
      1. Validate pack_id is mounted (find in packs[] by pack_id where mounted==1), return NT_ERR_INVALID_ARG if not found
      2. Validate blob_size >= sizeof(NtPackHeader), return NT_ERR_INVALID_ARG if too small
      3. Cast blob to `const NtPackHeader *h`
      4. Validate h->magic == NT_PACK_MAGIC, return NT_ERR_INVALID_ARG
      5. Validate h->version <= NT_PACK_VERSION_MAX, return NT_ERR_INVALID_ARG
      6. Validate h->header_size <= blob_size, return NT_ERR_INVALID_ARG
      7. Validate h->total_size == blob_size, return NT_ERR_INVALID_ARG
      8. Compute CRC32: `uint32_t data_size = blob_size - h->header_size; uint32_t computed = nt_crc32(blob + h->header_size, data_size); if (computed != h->checksum) return NT_ERR_INVALID_ARG;`
      9. If data_size == 0 and header has 0 assets, CRC32 of zero-length data must still match checksum field
      10. Cast entries: `const NtAssetEntry *entries = (const NtAssetEntry *)(blob + sizeof(NtPackHeader));`
      11. For each entry [0..h->asset_count): find next free slot in assets[] (asset_count < NT_RESOURCE_MAX_ASSETS), populate NtAssetMeta fields from NtAssetEntry, set state = NT_ASSET_STATE_REGISTERED, set pack_index = index of this pack in packs[]
      12. Update pack's asset_start and asset_count
      13. Store blob pointer and blob_size in PackMeta
      14. Set needs_resolve = true
      15. Return NT_OK
    - `nt_resource_step()`: For now, stub that clears needs_resolve flag (full resolve logic in Plan 02)
    - Slot pool helpers (static): `resource_make(index, gen)`, `slot_alloc()`, `slot_free(index)` following entity pattern with generation skip-zero
    - Stub implementations for mount/unmount/set_priority/request/get/is_ready/get_state/create_pack/register/unregister/set_placeholder -- mount creates PackMeta entry, other stubs return NT_ERR_INVALID_ARG or 0/false as appropriate. mount() is needed by parse_pack() in tests.
    - mount() implementation: find free slot in packs[] (mounted==0), set pack_id, priority, pack_type=NT_PACK_FILE, mounted=1, return NT_OK. Return NT_ERR_INVALID_ARG if no free slot.
